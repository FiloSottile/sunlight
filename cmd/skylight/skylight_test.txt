# "mirror" is not itself an origin: the more specific mirror pattern wins,
# redirecting to a path that is then a 404.
GET https://witness.example.org/mirror/checkpoint
redirect == /mirror/checkpoint/

# Path traversal is stopped at three independent layers: ServeMux cleans
# literal dot-dot segments into a redirect, StripPrefix rejects requests whose
# escaped path doesn't literally carry the origin prefix, and os.Root confines
# the file server.

GET https://witness.example.org/../checkpoint
redirect == /checkpoint/

GET https://witness.example.org/%2e%2e/checkpoint
code == 404

GET https://witness.example.org/..%2f../checkpoint
code == 404

GET https://witness.example.org/a%2Fb/checkpoint
code == 404

GET https://witness.example.org/mirror/..%2fmirror%2f000de184301123b9278c086a7d9f68c9577e83872224c197058fd90784ea9706/checkpoint
code == 404

# The home redirect and /health apply to any host without a more specific

# An override of the default test User-Agent, minting the anonymous client
# classification checked below. The checkpoint endpoint is not rate limited,
# so this doesn't draw down the anonymous budget.
GET https://rome2026h1.example.org/checkpoint
reqheader User-Agent skylight-webtest-anonymous
body contains rome2026h1.example.org

# /metrics runs last, checking the labels minted by the requests above: the
# kind classification of each endpoint, the per-log and per-origin log labels,
# the User-Agent client classification, and that file server errors are
